GENERAL

FEW-SHOT EXAMPLES


Q1: What percent of orders are shipped vs picked up?


Plan:
  Table: transactions
  Columns: ORDER_ID, DELIVERY_METHOD
  Probe: DISTINCT DELIVERY_METHOD → ['SHIP', 'PICKUP', 'DELIVERY'] (example)


Final SQL:
WITH order_level AS (
  SELECT
    ORDER_ID,
    ANY_VALUE(DELIVERY_METHOD) AS DELIVERY_METHOD
  FROM `lakshya-agenticai.retail.transaction_data`
  GROUP BY ORDER_ID
)
SELECT
  DELIVERY_METHOD,
  COUNT(*)                                              AS num_orders,
  ROUND(100 * COUNT(*) / SUM(COUNT(*)) OVER (), 2)     AS percent_orders
FROM order_level
GROUP BY DELIVERY_METHOD
ORDER BY percent_orders DESC;


Q2: Top selling categories by revenue


Final SQL:
SELECT
  PRODUCT_CATEGORY,
  ROUND(SUM(ITEM_QUANTITY * ITEM_REVENUE), 2)           AS total_revenue,
  ROUND(
    100 * SUM(ITEM_QUANTITY * ITEM_REVENUE)
        / SUM(SUM(ITEM_QUANTITY * ITEM_REVENUE)) OVER (),
    2
  )                                                      AS revenue_percent
FROM `lakshya-agenticai.retail.transaction_data`
WHERE IS_RETURN = FALSE
  AND IS_CANCELLATION = FALSE
GROUP BY PRODUCT_CATEGORY
ORDER BY total_revenue DESC;


Q3: Top 5 highest-selling products within the men's category


Plan:
  Table: transactions
  Columns: PRODUCT_ID, PRODUCT_CATEGORY, ITEM_QUANTITY, ITEM_REVENUE
  Probe DISTINCT PRODUCT_CATEGORY → values include "MEN'S", "Men's", "MENS" (unknown)
  Strategy: REGEXP_CONTAINS(LOWER(...), r"\bmen'?s?\b") covers all variants safely.


Final SQL:
SELECT
  PRODUCT_ID,
  ANY_VALUE(PRODUCT_CATEGORY)                          AS product_category,
  ROUND(SUM(ITEM_QUANTITY * ITEM_REVENUE), 2)          AS total_revenue
FROM `lakshya-agenticai.retail.transaction_data`
WHERE IS_RETURN = FALSE
  AND IS_CANCELLATION = FALSE
  AND REGEXP_CONTAINS(LOWER(PRODUCT_CATEGORY), r"\bmen'?s?\b")
GROUP BY PRODUCT_ID
ORDER BY total_revenue DESC
LIMIT 5;


Q4: Average return rate, total revenue, and average order value per purchase channel

Plan:
  Table: transactions
  Columns: PURCHASE_CHANNEL, ORDER_ID, ITEM_QUANTITY, ITEM_REVENUE, IS_RETURN
  Note: Return rate is computed across ALL orders (including returns).
        Revenue and AOV exclude returns and cancellations.


Final SQL:
WITH order_level AS (
  SELECT
    PURCHASE_CHANNEL,
    ORDER_ID,
    ROUND(SUM(ITEM_QUANTITY * ITEM_REVENUE), 2)             AS order_revenue,
    MAX(CASE WHEN IS_RETURN = TRUE THEN 1 ELSE 0 END)       AS is_return_order,
    MAX(CASE WHEN IS_CANCELLATION = TRUE THEN 1 ELSE 0 END) AS is_cancel_order
  FROM `lakshya-agenticai.retail.transaction_data`
  GROUP BY PURCHASE_CHANNEL, ORDER_ID
)
SELECT
  PURCHASE_CHANNEL,
  ROUND(SUM(CASE WHEN is_cancel_order = 0 AND is_return_order = 0
                 THEN order_revenue END), 2)               AS total_revenue,
  ROUND(AVG(CASE WHEN is_cancel_order = 0 AND is_return_order = 0
                 THEN order_revenue END), 2)               AS avg_order_value,
  ROUND(100 * AVG(is_return_order), 2)                     AS return_rate_percent
FROM order_level
GROUP BY PURCHASE_CHANNEL
ORDER BY total_revenue DESC;

Q6: Return and cancellation rates month over month


Final SQL:
WITH order_level AS (
  SELECT
    FORMAT_DATE('%Y-%m', DATE(ORDER_DATETIME))              AS order_month,
    ORDER_ID,
    MAX(CASE WHEN IS_RETURN = TRUE      THEN 1 ELSE 0 END)  AS is_return_order,
    MAX(CASE WHEN IS_CANCELLATION = TRUE THEN 1 ELSE 0 END) AS is_cancel_order
  FROM `lakshya-agenticai.retail.transaction_data`
  GROUP BY order_month, ORDER_ID
)
SELECT
  order_month,
  COUNT(*)                                             AS total_orders,
  ROUND(100 * AVG(is_return_order), 2)                AS return_rate_percent,
  ROUND(100 * AVG(is_cancel_order), 2)                AS cancellation_rate_percent
FROM order_level
GROUP BY order_month
ORDER BY order_month;

Q8: What is the return rate per purchase channel?


Plan:
  Table: transactions
  Columns: PURCHASE_CHANNEL, ORDER_ID, IS_RETURN
  Probe: DISTINCT PURCHASE_CHANNEL → e.g. ['IN-STORE', 'ONLINE', 'MOBILE']
  Note: Return rate must be computed across ALL orders (including returns),
        so no IS_RETURN filter at the outer level.


Final SQL:
WITH order_level AS (
  SELECT
    PURCHASE_CHANNEL,
    ORDER_ID,
    MAX(CASE WHEN IS_RETURN = TRUE THEN 1 ELSE 0 END)  AS is_return_order
  FROM `lakshya-agenticai.retail.transaction_data`
  GROUP BY PURCHASE_CHANNEL, ORDER_ID
)
SELECT
  PURCHASE_CHANNEL,
  COUNT(*)                                        AS total_orders,
  SUM(is_return_order)                            AS returned_orders,
  ROUND(100 * AVG(is_return_order), 2)            AS return_rate_percent
FROM order_level
GROUP BY PURCHASE_CHANNEL
ORDER BY return_rate_percent DESC;

Q9: How does the day-wise session trend across the month?

SELECT
  count(SESSION_UUID) as unique_sessions,
  EXTRACT(DAY FROM DATE_FILTER) AS Day,
FROM `lakshya-agenticai.retail.clickstream_session_data`
GROUP BY day order by day

Q10: What is the day-wise trend of cart, checkout, and order activity?

SELECT
  DATE(SESSION_TIMESTAMP_NY)        AS date,
  SUM(CART_FLAG)                    AS total_cart,
  SUM(CHECKOUT_FLAG)                AS total_checkout,
  SUM(ORDER_FLAG)                   AS total_orders
FROM `lakshya-agenticai.retail.clickstream_session_data`
GROUP BY date
ORDER BY date

Q11: What is the day-wise cart-to-order and checkout-to-order conversion rate?

SELECT
  date,
  total_cart,
  total_checkout,
  total_orders,
  ROUND(SAFE_DIVIDE(total_orders, total_cart)     * 100, 2) AS cart_to_order_pct,
  ROUND(SAFE_DIVIDE(total_orders, total_checkout) * 100, 2) AS checkout_to_order_pct
FROM (
  SELECT
    DATE(SESSION_TIMESTAMP_NY)  AS date,
    SUM(CART_FLAG)              AS total_cart,
    SUM(CHECKOUT_FLAG)          AS total_checkout,
    SUM(ORDER_FLAG)             AS total_orders
  FROM `lakshya-agenticai.retail.clickstream_session_data`
  GROUP BY date
)
ORDER BY date

Q12: What is the session share by device type and operating system?

SELECT
  DEVICE_TYPE,
  OPERATING_SYSTEM,
  COUNT(DISTINCT SESSION_UUID)                                                        AS unique_sessions,
  ROUND(COUNT(DISTINCT SESSION_UUID) * 100.0 / SUM(COUNT(DISTINCT SESSION_UUID)) OVER(), 2) AS pct_of_total
FROM `lakshya-agenticai.retail.clickstream_session_data`
where DEVICE_TYPE is not NULL and OPERATING_SYSTEM is not NULL
GROUP BY DEVICE_TYPE, OPERATING_SYSTEM
ORDER BY  1,2
